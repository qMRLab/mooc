\[ \frac{d M_{y,f}}{dt}=-\frac{M_{y,f}}{T_{2,f}}+\Delta M_{x,f}-\text{Re}\left( \omega_{1} \right)M_{z,f} \]

\[ \frac{d M_{z,r}}{dt}=-\frac{\left( M_{0,r}-M_{z,r} \right)}{T_{1,r}}-k_{r}M_{z,r}+k_{f}M_{z,f}+WM_{z,r} \]
